44 MatrixN Sigmoid(const MatrixN& m) {
45 MatrixN mn(m);
46 MatrixN y;
47 // Standard sigmoid is not numerically stable (large -x instable)
48 //y=(1.0/(1.0+(mn.array() * -1.0).exp()));
49 // Alternative via tanh is stable and doesn't need case distinctions for large + - inf values of x.
50 // alt see: http://timvieira.github.io/blog/post/2014/02/11/exp-normalize-trick/
51 y=((mn.array()/2.0).tanh()-1.0)/2.0+1.0;
52 return y;
53 }
54 MatrixN dSigmoid(const MatrixN& y) {
55 return (y.array()*(y.array()-1.0) * -1.0).matrix();
56 }

66 MatrixN dRelu(const MatrixN& x) {
67 MatrixN y=x;
68 for (unsigned int i=0; i<y.size(); i++) {
69 if (y(i)>0.0) y(i)=1.0;
70 else y(i)=0.0;
71 }
72 return y;
73 }
74 MatrixN Selu(const MatrixN& x) {
75 // "scaled exponential linear units" (SELUs), https://arxiv.org/abs/1706.02515
76 floatN alpha = 1.6732632423543772848170429916717;
77 floatN scale = 1.0507009873554804934193349852946;
78 MatrixN y=x;
79 // return scale*np.where(x>=0.0, x, alpha*np.exp(x)-alpha)
80 for (unsigned int i=0; i<(unsigned int)y.size(); i++) {
81 if (y(i)<0.0) y(i)=alpha*exp(y(i))-alpha;
82 }
83 y=y*scale;
84 return y;
85 }
86 MatrixN dSelu(const MatrixN& x) {
87 // "scaled exponential linear units" (SELUs), https://arxiv.org/abs/1706.02515
88 MatrixN y=x;
89 floatN alpha = 1.6732632423543772848170429916717;
90 floatN scale = 1.0507009873554804934193349852946;
91 for (unsigned int i=0; i<(unsigned int)y.size(); i++) {
92 if (y(i)>=0) y(i)=scale;
93 else y(i)=alpha*exp(y(i))*scale;
94 }
95 return y;
96 }
97 MatrixN Resilu(const MatrixN& x) {
98 // Resilu should combine a non-linearity (similar to relu) and a residual, linear connection
99 // r(x) = 1/(1-exp(-x/a)). Variing a limits in relu. r(x) can be rewritten as r(x)=x/(exp(x)-1) + x (the linear, additive residual part)
100 // Problem: phase-transition at 0, hence:
101 // approximate with taylor-expansion up to O(6) for -h<x<h
102 floatN h=0.001;
103 floatN x2;
104 MatrixN y=x.array()/(1.0-(x.array() * -1.0).exp());
105 for (unsigned int i=0; i<(unsigned int)x.size(); i++) {
106 if (x(i) < h && x(i) > -1.0 * h ) {
107 x2=x(i)*x(i);
108 y(i)=1.0+x(i)/2.0+x2/12.0-x2*x2/720.0+x2*x2*x2/30240.0;
109 }
110 }
111 return y;
112 }
113 MatrixN dResilu(const MatrixN& x) {
114 // approximate with taylor-expansion up to O(6) for -h<x<h
115 floatN h=0.001;
116 floatN x2;
117 MatrixN e=x.array().exp();
118 MatrixN e1 = e.array()-1.0;
119 MatrixN y = e.array()*(e.array()-x.array()-1.0)/e1.array()/e1.array();
120 for (unsigned int i=0; i<(unsigned int)x.size(); i++) {
121 if (x(i) < h && x(i) > -1.0 * h ) {
122 x2 = x(i)*x(i);
123 y(i)=0.5+x(i)/6.0-x2*x(i)/180.0+x2*x2*x(i)/5040.0-x2*x2*x2*x(i)/151200; // //x(i)/6.0+0.5819767068693265; // 0.58.. == 1/(e-1)
124 }
125 }
126 return y;
