125// Uses the Adam update rule, which incorporates moving averages of both the
126// gradient and its square and a bias correction term. [CS231]
127// - learning_rate: Scalar learning rate.
128// - beta1: Decay rate for moving average of first moment of gradient.
129// - beta2: Decay rate for moving average of second moment of gradient.
130// - epsilon: Small scalar used for smoothing to avoid dividing by zero.
134class Adam : public Optimizer {
135 floatN b1;
136 floatN b2;
137 floatN ep;
138public:
139 Adam(const json& jx) {
144 j=jx;
145 j["learning_rate"]=j.value("learning_rate", (floatN)1e-3);
146 lr=j["learning_rate"];
147 j["beta1"]=j.value("beta1", (floatN)0.9);
148 j["beta2"]=j.value("beta2", (floatN)0.999);
149 j["epsilon"]=j.value("epsilon", (floatN)1e-7);
150 }
151 virtual MatrixN update(MatrixN& w, MatrixN& dw, string wName, t_cppl* pOptimizerState) override {
160 b1=j["beta1"];
161 b2=j["beta2"];
162 ep=j["epsilon"];
163 string cName_m=wName+"-m";
164 if (pOptimizerState->find(cName_m)==pOptimizerState->end()) {
165 MatrixN z=MatrixN(w);
166 z.setZero();
167 cppl_update(pOptimizerState, cName_m, &z);
168 }
169 string cName_v=wName+"-v";
170 if (pOptimizerState->find(cName_v)==pOptimizerState->end()) {
171 MatrixN z=MatrixN(w);
172 z.setZero();
173 cppl_update(pOptimizerState, cName_v, &z);
174 }
175 string cName_t=wName+"-t";
176 if (pOptimizerState->find(cName_t)==pOptimizerState->end()) {
177 MatrixN z1(1,1);
178 z1.setZero();
179 cppl_update(pOptimizerState, cName_t, &z1);
180 }
181 floatN t=(*(*pOptimizerState)[cName_t])(0,0) + 1.0;
182 (*(*pOptimizerState)[cName_t])(0,0)=t;
183 *(*pOptimizerState)[cName_m]=b1 * (*(*pOptimizerState)[cName_m]) + (1.0 -b1) * dw;
184 *(*pOptimizerState)[cName_v]=b2 * (*(*pOptimizerState)[cName_v]).array() + (1.0 -b2) * (dw.array() * dw.array());
185 MatrixN mc = 1.0/(1.0-pow(b1,t)) * (*(*pOptimizerState)[cName_m]);
186 MatrixN vc = 1.0/(1.0-pow(b2,t)) * (*(*pOptimizerState)[cName_v]);
187 w = w.array() - lr * mc.array() / (vc.array().sqrt() + ep);
188 return w;
189 }
