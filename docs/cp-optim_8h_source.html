122// Uses the Adam update rule, which incorporates moving averages of both the
123// gradient and its square and a bias correction term. [CS231]
124// - learning_rate: Scalar learning rate.
125// - beta1: Decay rate for moving average of first moment of gradient.
126// - beta2: Decay rate for moving average of second moment of gradient.
127// - epsilon: Small scalar used for smoothing to avoid dividing by zero.

135 floatN ep;

154 b1=j.value("beta1", (floatN)0.9);
155 b2=j.value("beta2", (floatN)0.999);
156 ep=j.value("epsilon", (floatN)1e-7);
157 string cName_m=wName+"-m";
158 if (pStateCache->find(cName_m)==pStateCache->end()) {
159 MatrixN z=MatrixN(w);
160 z.setZero();
161 cppl_update(pStateCache, cName_m, &z);
162 }
163 string cName_v=wName+"-v";
164 if (pStateCache->find(cName_v)==pStateCache->end()) {
165 MatrixN z=MatrixN(w);
166 z.setZero();
167 cppl_update(pStateCache, cName_v, &z);
168 }
169 string cName_t=wName+"-t";
170 if (pStateCache->find(cName_t)==pStateCache->end()) {
171 MatrixN z1(1,1);
172 z1.setZero();
173 cppl_update(pStateCache, cName_t, &z1);
174 }
175 floatN t=(*(*pStateCache)[cName_t])(0,0) + 1.0;
176 (*(*pStateCache)[cName_t])(0,0)=t;
177 *(*pStateCache)[cName_m]=b1 * (*(*pStateCache)[cName_m]) + (1.0 -b1) * dw;
178 *(*pStateCache)[cName_v]=b2 * (*(*pStateCache)[cName_v]).array() + (1.0 -b2) * (dw.array() * dw.array());
179 MatrixN mc = 1.0/(1.0-pow(b1,t)) * (*(*pStateCache)[cName_m]);
180 MatrixN vc = 1.0/(1.0-pow(b2,t)) * (*(*pStateCache)[cName_v]);
