65 virtual MatrixN forward(const MatrixN& x, t_cppl* pcache, t_cppl* pstates, int id=0) override {
66 // XXX cache x2c and use allocated memory for im2col call!
67 auto N=shape(x)[0];
68 if (shape(x)[1]!=(unsigned int)C*W*H) {
69 cerr << "PoolFw: Invalid input data x: expected C*H*W=" << C*H*W << ", got: " << shape(x)[1] << endl;
70 return MatrixN(0,0);
71 }
72 MatrixN *pmask = new MatrixN(N, C*H*W);
73 pmask->setZero();
74 MatrixN y(N,C*WO*HO);
75 y.setZero();
76 floatN mx;
77 int chw,chowo,px0,iy0;
78 int xs, ys,px, mxi, myi;
79 for (int n=0; n<(int)N; n++) {
80 for (int c=0; c<C; c++) {
81 chw=c*H*W;
82 chowo=c*HO*WO;
83 for (int iy=0; iy<HO; iy++) {
84 iy0=chowo+iy*WO;
85 for (int ix=0; ix<WO; ix++) {
86 mx=0.0; mxi= (-1); myi= (-1);
87 for (int cy=0; cy<HH; cy++) {
88 ys=iy*stride+cy;
89 px0=chw+ys*W;
90 if (ys>=H) continue;
91 for (int cx=0; cx<WW; cx++) {
92 xs=ix*stride+cx;
93 if (xs>=W) continue;
94 px=px0+xs;
95 if (cx==0 && cy==0) {
96 mx=x(n,px);
97 myi=n;
98 mxi=px;
99 } else {
100 if (x(n,px)>mx) {
101 mx=x(n,px);
102 myi=n;
103 mxi=px;
104 }
105 }
106 }
107 }
108 y(n,iy0+ix)=mx;
109 if (mxi!=(-1) && myi!=(-1)) (*pmask)(myi,mxi) = 1.0;
110 }
111 }
112 }
113 }
114
115 //cerr << "x:" << endl << x << endl;
116 //cerr << "mask:" << endl << *pmask << endl;
117
118 if (pcache!=nullptr) cppl_set(pcache, "x", new MatrixN(x)); // XXX where do we need x?
119 if (pcache!=nullptr) cppl_set(pcache, "mask", pmask);
120
121 if (pcache==nullptr) delete pmask;
122 return y;

130 MatrixN dx(N,C*W*H);
131 dx.setZero();
132 int chw,chowo,px0,py0,ix0;
133 int xs, ys, px, py;
134 for (int n=0; n<(int)N; n++) {
135 for (int c=0; c<C; c++) {
136 chw=c*H*W;
137 chowo=c*WO*HO;
138 for (int iy=0; iy<HO; iy++) {
139 py0=chowo+iy*WO;
140 for (int ix=0; ix<WO; ix++) {
141 ix0=ix*stride;
142 for (int cy=0; cy<HH; cy++) {
143 ys=iy*stride+cy;
144 px0=chw+ys*W;
145 if (ys>=H) continue;
146 for (int cx=0; cx<WW; cx++) {
147 xs=ix0+cx;
148 if (xs>=W) continue;
149 px=px0+xs;
150 py=py0+ix;
151 MatrixN *pmask=(*pcache)["mask"];
152 dx(n, px) += dchain(n,py) * (*pmask)(n, px);
153 }
154 }
155 }
156 }
157 }
158 }
