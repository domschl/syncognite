52 /*
53 Forward pass for a temporal affine layer. The input is a set of D-dimensional
54 vectors arranged into a minibatch of N timeseries, each of length T. We use
55 an affine function to transform each of those vectors into a new vector of
56 dimension M.
57
58 Inputs:
59 - x: Input data of shape (N, {T * D})
60 - w: Weights of shape (D, M)
61 - b: Biases of shape (M,)
62
63 Returns a tuple of:
64 - out: Output data of shape (N, {T * M})
65 - cache: Values needed for the backward pass
66 """
67 */
68 virtual MatrixN forward(const MatrixN& x, t_cppl* pcache, t_cppl* pstates, int id=0) override {
69 if (x.cols() != T*D) {
70 cerr << layerName << ": " << "Forward: dimension mismatch TemporalAFfine in x*W: x(cols):" << x.cols() << " T*D:" << T*D << endl;
71 MatrixN y(0,0);
72 return y;
73 }
74 if (pcache!=nullptr) cppl_set(pcache, "x", new MatrixN(x));
75 int N=(int)x.rows();
76 // x: [N, (T * D)] -> [(N * T), D]
77 MatrixN xt(N*T, D);
78 for (int n=0; n<N; n++) {
79 for (int t=0; t<T; t++) {
80 for (int d=0; d<D; d++) {
81 xt(n*T+t,d)=x(n,t*D+d);
82 }
83 }
84 }
85 if (pcache!=nullptr) cppl_set(pcache, "xt", new MatrixN(xt));
86 MatrixN yt=(xt * (*params["W"])).rowwise() + RowVectorN(*params["b"]);
87
88 MatrixN y(N,T*M);
89 for (int n=0; n<N; n++) {
90 for (int t=0; t<T; t++) {
91 for (int m=0; m<M; m++) {
92 y(n,t*M+m)=yt(n*T+t,m);
93 }
94 }
95 }
96
97 return y;
98 }
99 /*
100 """

102
103 Input:
104 - dout: Upstream gradients of shape (N, T, M)
105 - cache: Values from forward pass
106
107 Returns a tuple of:

116 for (int n=0; n<N; n++) {
117 for (int t=0; t<T; t++) {
118 for (int m=0; m<M; m++) {
119 dchaint(n*T+t,m)=dchain(n,t*M+m);
120 }
121 }
122 }
123
124 MatrixN dxt = dchaint * (*params["W"]).transpose(); // dx
125 if ((*(*pcache)["xt"]).transpose().cols() != dchaint.rows()) {
126 cerr << endl << "TA: xt.T dot dchaint:" << shape((*(*pcache)["xt"]).transpose()) << ", " << shape(dchaint) << endl;
127 exit(-1);
128 }
129 cppl_set(pgrads, "W", new MatrixN((*(*pcache)["xt"]).transpose() * dchaint)); //dW
130 cppl_set(pgrads, "b", new MatrixN(dchaint.colwise().sum())); //db
131 MatrixN dx(N, T*D);
132 for (int n=0; n<N; n++) {
133 for (int t=0; t<T; t++) {
134 for (int d=0; d<D; d++) {
135 dx(n,t*D+d)=dxt(n*T+t,d);
136 }
137 }
138 }
139 return dx;
140 }
