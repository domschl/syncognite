80 MatrixN Sigmoid(const MatrixN& m) {
81 MatrixN mn(m);
82 MatrixN y;
83 // Standard sigmoid is not numerically stable (large -x instable)
84 //y=(1.0/(1.0+(mn.array() * -1.0).exp()));
85 // Alternative via tanh is stable and doesn't need case distinctions for large + - inf values of x.
86 // alt see: http://timvieira.github.io/blog/post/2014/02/11/exp-normalize-trick/
87 y=((mn.array()/2.0).tanh()-1.0)/2.0+1.0;
88 return y;
89 }
90 /* Forward pass for a single timestep of an LSTM. [CS231]
91 The input data has dimension D, the hidden state has dimension H, and we
92 use a minibatch size of N.
93 Inputs:
94 - x: Input data, of shape (N, D)
95 - hprev: Previous hidden state, of shape (N, H)
96 - cprev: previous cell state, of shape (N, H)
97 - Wxh: Input-to-hidden weights, of shape (D, 4H)
98 - Whh: Hidden-to-hidden weights, of shape (H, 4H)
99 - bh: Biases, of shape (4H,)
100 Returns a tuple of:
101 - hnext: Next hidden state, of shape (N, H)
102 - cnext: Next cell state, of shape (N, H)
103 */
104 t_cppl forward_step(const MatrixN& x, t_cppl* pcache, t_cppl* pstates, int id=0) {
105 t_cppl cp;
106 int N=shape(x)[0];
107 MatrixN hprev = *(*pstates)[hname];
108 MatrixN cprev = *(*pstates)[cname];
109 MatrixN xhx = (hprev * *(params["Whh"]) + x * *(params["Wxh"])).rowwise() + RowVectorN(*params["bh"]);
110 MatrixN i = Sigmoid(xhx.block(0,0,N,H));
111 MatrixN f = Sigmoid(xhx.block(0,H,N,H));
112 MatrixN o = Sigmoid(xhx.block(0,2*H,N,H));
113 MatrixN g = (xhx.block(0,3*H,N,H)).array().tanh();
114 MatrixN cnext = f.array() * cprev.array() + i.array() * g.array();
115 cp[cname0]=new MatrixN(cnext);
116 MatrixN tnc=cnext.array().tanh();
117 MatrixN hnext=tnc.array() * o.array();
118 cp[hname0]=new MatrixN(hnext);
119 if (pcache!=nullptr) {
120 cppl_set(pcache,"tnc",new MatrixN(tnc));
121 cppl_set(pcache,"xhx",new MatrixN(xhx));
122 cppl_set(pcache,hname,new MatrixN(hprev));
123 cppl_set(pcache,cname,new MatrixN(cprev));
124 cppl_set(pcache,"i",new MatrixN(i));
125 cppl_set(pcache,"f",new MatrixN(f));
126 cppl_set(pcache,"o",new MatrixN(o));
127 cppl_set(pcache,"g",new MatrixN(g));
128 cppl_set(pcache,"x",new MatrixN(x));
129 }
130 return cp;
131 }

133 /* Backward pass for a single timestep of an LSTM.
134 Inputs:
135 - dnext_h: Gradients of next hidden state, of shape (N, H)
136 - dnext_c: Gradients of next cell state, of shape (N, H)
137 Returns a tuple of:
138 - dx: Gradient of input data, of shape (N, D)
139 - dprev_h: Gradient of previous hidden state, of shape (N, H)
140 - dprev_c: Gradient of previous cell state, of shape (N, H)
141 - dWx: Gradient of input-to-hidden weights, of shape (D, 4H)
142 - dWh: Gradient of hidden-to-hidden weights, of shape (H, 4H)
143 - db: Gradient of biases, of shape (4H,) */
144 MatrixN backward_step(t_cppl& cp, t_cppl* pcache, t_cppl* pstates, t_cppl* pgrads, int id=0) {
145 MatrixN hprev = *(*pcache)[hname];
146 MatrixN cprev = *(*pcache)[cname];
147 MatrixN tnc=*(*pcache)["tnc"];
148 MatrixN xhx=*(*pcache)["xhx"];
149 MatrixN i=*(*pcache)["i"];
150 MatrixN f=*(*pcache)["f"];
151 MatrixN o=*(*pcache)["o"];
152 MatrixN g=*(*pcache)["g"];
153 MatrixN x=*(*pcache)["x"];
154 MatrixN dhnext=*cp[hname]; // dhnext
155 MatrixN dcnext;
156 if (cp.find(cname)!=cp.end()) {
157 dcnext=*cp[cname]; // dcnext
158 } else {
159 dcnext=*cp[hname]; // zero-fake init
160 dcnext.setZero(); // Seems to be CS231 api-shortcut of backward() not accomodating dc.
161 }
162
163 MatrixN ddo=tnc.array() * dhnext.array();
164 MatrixN dtnc=o.array() * dhnext.array();
165 dcnext = dcnext + ((1.0 - tnc.array() * tnc.array()) * dtnc.array()).matrix();
166 MatrixN df = cprev.array() * dcnext.array();
167 MatrixN dcprev = f.array() * dcnext.array();
168 MatrixN di = g.array() * dcnext.array();
169 MatrixN dg = i.array() * dcnext.array();
170 MatrixN dps(N,4*H);
171 MatrixN dps3 = (g.array() * g.array() - 1.0) * -1.0 * dg.array();
172 MatrixN dps2 = (o.array() - 1.0) * -1.0 * o.array() * ddo.array();
173 MatrixN dps1 = (f.array() -1.0) * -1.0 * f.array() * df.array();
174 MatrixN dps0 = (i.array() - 1.0) * -1.0 * i.array() * di.array();
175 dps << dps0, dps1, dps2, dps3; // stack matrices
176 MatrixN dphh = dps;
177 MatrixN dpshx = dps;
178 MatrixN dphx = dpshx.transpose();
179 MatrixN dbh = dpshx.colwise().sum();
180 MatrixN dx = ((*params["Wxh"]) * dphx).transpose();
181 MatrixN dWxh = (dphx * x).transpose();
182 MatrixN dhprev = ((*params["Whh"]) * dphh.transpose()).transpose();
183 MatrixN dWhh = hprev.transpose() * dphh;
184
185 (*pgrads)["Wxh"] = new MatrixN(dWxh);
186 (*pgrads)["Whh"] = new MatrixN(dWhh);
187 (*pgrads)["bh"] = new MatrixN(dbh);
188 (*pgrads)[hname0] = new MatrixN(dhprev);
189 (*pgrads)[cname0] = new MatrixN(dcprev);
190
191 if (maxClip!=0.0) {
192 dx = dx.array().min(maxClip).max(-1*maxClip);
193 *(*pgrads)["Wxh"] = (*(*pgrads)["Wxh"]).array().min(maxClip).max(-1*maxClip);
194 *(*pgrads)["Whh"] = (*(*pgrads)["Whh"]).array().min(maxClip).max(-1*maxClip);
195 *(*pgrads)["bh"] = (*(*pgrads)["bh"]).array().min(maxClip).max(-1*maxClip);
196 *(*pgrads)[hname0] = (*(*pgrads)[hname0]).array().min(maxClip).max(-1*maxClip);
197 *(*pgrads)[cname0] = (*(*pgrads)[cname0]).array().min(maxClip).max(-1*maxClip);
198 }
199 return dx;
200 }

202 MatrixN tensorchunk(const MatrixN& x, vector<int>dims, int b) {
203 int A=dims[0];
204 int C=dims[2];
205 MatrixN xi(A,C);
206 int a,c;
207 for (a=0; a<A; a++) {
208 for (c=0; c<C; c++) {
209 xi(a,c) = x(a,b*C+c);
210 }
211 }
212 return xi;

225 virtual MatrixN forward(const MatrixN& x, t_cppl* pcache, t_cppl* pstates, int id=0) override {
226 MatrixN hn(N,T*H);
227 MatrixN cn(N,T*H);
228
229 if (x.cols() != D*T) {
230 cerr << layerName << ": " << "Forward: dimension mismatch in x:" << shape(x) << " D*T:" << D*T << ", D:" << D << ", T:" << T << ", H:" << H << endl;
231 MatrixN h(0,0);
232 return h;
233 }
234 int N=shape(x)[0];
235
236 if (pstates->find(hname)==pstates->end()) {
237 MatrixN *ph= new MatrixN(N,H);
238 ph->setZero();
239 cppl_set(pstates, hname, ph);
240 }
241 if (pstates->find(cname)==pstates->end()) {
242 MatrixN *pc= new MatrixN(N,H);
243 pc->setZero();
244 cppl_set(pstates, cname, pc);
245 }
246
247 MatrixN ht=*(*pstates)[hname];
248 MatrixN ct=*(*pstates)[cname];
249
250 for (int t=0; t<T; t++) {
251 t_cppl cache{};
252 t_cppl states{};
253 states[hname]=&ht;
254 states[cname]=&ct;
255 MatrixN xi=tensorchunk(x,{N,T,D},t);
256 t_cppl cp = forward_step(xi,&cache,&states, id);
257 ht = *cp[hname0];
258 ct = *cp[cname0];
259 cppl_delete(&cp);
260 tensorchunkinsert(&hn, ht, {N,T,H}, t);
261 tensorchunkinsert(&cn, ct, {N,T,H}, t);
262 if (pcache!=nullptr) {
263 string name="t"+std::to_string(t);
264 mlPush(name,&cache,pcache);
265 } else cppl_delete(&cache);
266 }
267 cppl_update(pstates,hname,&ht);
268 cppl_update(pstates,cname,&ct);
269 if (hn.cols() != (T*H)) {
270 cerr << "Inconsistent LSTM-forward result: " << shape(hn) << endl;
271 }
272 return hn;
273 }

275 virtual MatrixN backward(const MatrixN& dchain, t_cppl* pcache, t_cppl* pstates, t_cppl* pgrads, int id=0) override {
276 MatrixN dx(N,T*D);
277
278 MatrixN dWxh,dWhh,dbh;
279 string name;
280 int N=shape(dchain)[0];
281 MatrixN dhi,dci,dxi,dphi,dpci;
282 dx.setZero();
283 for (int t=T-1; t>=0; t--) {
284 t_cppl cache{};
285 t_cppl grads{};
286 t_cppl states{};
287 if (t==T-1) {
288 dhi=tensorchunk(dchain,{N,T,H},t);
289 dci=dhi; // That should also be set by dchain!!
290 dci.setZero();
291 } else {
292 dhi=tensorchunk(dchain,{N,T,H},t) + dphi;
293 dci=dpci;
294 }
295 name="t"+std::to_string(t);
296 mlPop(name,pcache,&cache);
297 t_cppl cp;
298 cp[hname]=new MatrixN(dhi);
299 cp[cname]=new MatrixN(dci);
300 // cp[cname]=new MatrixN(dci); // XXX: dchain should also take care of dc! -> zero-init for now.
301 dxi=backward_step(cp,&cache,&states, &grads,id);
302 cppl_delete(&cp);
303 tensorchunkinsert(&dx, dxi, {N,T,D}, t);
304 dphi=*grads[hname0];
305 dpci=*grads[cname0];
306 if (t==T-1) {
307 dWxh=*grads["Wxh"];
308 dWhh=*grads["Whh"];
309 dbh=*grads["bh"];
310 } else {
311 dWxh+=*grads["Wxh"];
312 dWhh+=*grads["Whh"];
313 dbh+=*grads["bh"];
314 }
315 cppl_delete(&grads);
316 }
317 (*pgrads)["Wxh"] = new MatrixN(dWxh);
318 (*pgrads)["Whh"] = new MatrixN(dWhh);
319 (*pgrads)["bh"] = new MatrixN(dbh);
320 (*pgrads)[hname0] = new MatrixN(dphi);
321 //(*pgrads)[cname0] = new MatrixN(dpci);
322
323 return dx;
324 }
