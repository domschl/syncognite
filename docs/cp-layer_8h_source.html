20MatrixN xavierInit(const MatrixN &w, XavierMode xavMode=XavierMode::XAV_STANDARD, floatN initfactor=1.0) {
21 if (initfactor == 0.0) {
22 cerr << "Initfactor=0.0 is invalid! Setting to 1.0" << endl;
23 initfactor=1.0;
24 }
25 floatN xavier = 2.0/std::sqrt((floatN)(w.cols()+w.rows())) * initfactor;
26 float mean=0.0;
27 float std=xavier / 2.0;
28 MatrixN wo(w);
29 MatrixN wot(w);
30 std::default_random_engine rde;
31 std::normal_distribution<float> distn(mean, std);
32 switch (xavMode) {
33 case XavierMode::XAV_NORMAL:
34 for (int i=0; i<wo.size(); i++) wo(i)=distn(rde);
35 break;
36 case XavierMode::XAV_ORTHONORMAL:
37 for (int i=0; i<wot.size(); i++) wot(i)=distn(rde);
38 if (wot.rows() == wot.cols()) {
39 wo = wot.householderQr().householderQ();
40 } else {
41 wo=wot; // we can only orthonormalize square matrices!
42 }
43 break;
44 case XavierMode::XAV_ORTHOGONAL:
45 for (int i=0; i<wot.size(); i++) wot(i)=distn(rde);
46 if (wot.rows() == wot.cols()) {
47 wo = wot.householderQr().householderQ();
48 wo *= xavier; // orthogonal instead of orthonormal!
49 } else {
50 wo=wot; // we can only orthonormalize square matrices!
51 }
52 break;
53 case XavierMode::XAV_STANDARD:
54 default:
55 wo.setRandom(); // [-1,1]
56 wo = wo * xavier/2.0; // (setRandom is [-1,1]-> fakt 0.5, xavier is 2/(ni+no))
57 break;
58 }
59 return wo;
60}
