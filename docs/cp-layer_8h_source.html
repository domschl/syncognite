28MatrixN xavierInit(const MatrixN &w, XavierMode xavierMode=XavierMode::XAV_STANDARD, floatN initfactor=1.0) {
36 if (initfactor == 0.0) {
37 cerr << "Initfactor=0.0 is invalid! Setting to 1.0" << endl;
38 initfactor=1.0;
39 }
40 floatN xavier = 2.0/std::sqrt((floatN)(w.cols()+w.rows())) * initfactor;
41 float mean=0.0;
42 float std=xavier / 2.0;
43 MatrixN wo(w);
44 MatrixN wot(w);
45 std::default_random_engine rde;
46 std::normal_distribution<float> distn(mean, std);
47 switch (xavierMode) {
48 case XavierMode::XAV_NORMAL:
49 for (int i=0; i<wo.size(); i++) wo(i)=distn(rde);
50 break;
51 case XavierMode::XAV_ORTHONORMAL:
52 for (int i=0; i<wot.size(); i++) wot(i)=distn(rde);
53 if (wot.rows() == wot.cols()) {
54 wo = wot.householderQr().householderQ();
55 } else {
56 wo=wot; // we can only orthonormalize square matrices!
57 }
58 break;
59 case XavierMode::XAV_ORTHOGONAL:
60 for (int i=0; i<wot.size(); i++) wot(i)=distn(rde);
61 if (wot.rows() == wot.cols()) {
62 wo = wot.householderQr().householderQ();
63 wo *= xavier; // orthogonal instead of orthonormal!
64 } else {
65 wo=wot; // we can only orthonormalize square matrices!
66 }
67 break;
68 case XavierMode::XAV_STANDARD:
69 default:
70 wo.setRandom(); // [-1,1]
71 wo = wo * xavier/2.0; // (setRandom is [-1,1]-> fakt 0.5, xavier is 2/(ni+no))
72 break;
73 }
74 return wo;
75}
